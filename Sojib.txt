        async function trackDownload(hackId) {
            try {

                console.log('Error tracking download: ' + error.message);
            }

        function isExpired(expiryDate) {
            if (!expiryDate) return false;
            
            const today = new Date().toISOString().split('T')[0];
            const result = expiryDate < today;
            return result;
        }
        
        function getDaysRemaining(expiryDate) {
            if (!expiryDate) return 0;
            
            const today = new Date();
            const expiry = new Date(expiryDate);
            const diffTime = expiry - today;
            const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
            
            return diffDays > 0 ? diffDays : 0;
        }
        
        // =========== SESSION MANAGEMENT ===========
        function initSession() {
            const lastSession = localStorage.getItem('lastSessionData');
            if (lastSession) {
                const sessionData = JSON.parse(lastSession);
                const timeSinceLastActivity = Date.now() - sessionData.lastActivity;
                
                if (timeSinceLastActivity > 1000 * 60 * 30) { // 30 minutes
                    resetProgressSilently();
                } else {
                    adsWatched = sessionData.adsWatched || 0;
                    sessionActive = true;
                    
                    if (adsWatched >= totalAds) {
                        // If already completed ads, show waiting screen
                        document.getElementById('adSection').style.display = 'none';
                        startWaitingPeriod();
                        waitingSeconds = 1; // Start from 1 second
                    }
                }
            }
            
            updateAdDisplay();
            saveSession();
        }
        
        function saveSession() {
            const sessionData = {
                adsWatched: adsWatched,
                lastActivity: Date.now()
            };
            localStorage.setItem('lastSessionData', JSON.stringify(sessionData));
        }
        
        function resetProgressSilently() {
            adsWatched = 0;
            sessionActive = false;
            waitingSeconds = 2;
            updateAdDisplay();
            saveSession();
            document.getElementById('filesSection').style.display = 'none';
            document.getElementById('waitingScreen').style.display = 'none';
            document.getElementById('adSection').style.display = 'block';
            
            // Clear timers
            clearInterval(waitingTimerInterval);
        }

        function updateAdDisplay() {

            const progressPercent = (adsWatched / totalAds) * 100;
            
            document.getElementById('adCounter').textContent = remaining;
            document.getElementById('progressFill').style.width = progressPercent + '%';
            document.getElementById('progressText').textContent = `${adsWatched}/${totalAds} ads watched`;
            
            if (remaining > 0) {
                document.getElementById('counterText').textContent = `${remaining} ads remaining`;
            } else {
                document.getElementById('counterText').innerHTML = '<span style="color:#00ff00">‚úÖ Completed</span>';
            }
            
            const btn = document.getElementById('downloadBtn');
            if (adsWatched >= totalAds && sessionActive && waitingSeconds <= 0) {
                btn.disabled = false;
                btn.classList.add('active');
                btn.textContent = '‚úÖ UNLOCK ALL FILES';
            } else {
                btn.disabled = true;
                btn.classList.remove('active');
                btn.textContent = `üîí WATCH ${remaining} MORE AD${remaining > 1 ? 'S' : ''}`;
            }
        }

        function showFiles() {
            if (adsWatched >= totalAds && sessionActive && waitingSeconds <= 0) {
                document.getElementById('filesSection').style.display = 'block';
                document.getElementById('downloadBtn').style.display = 'none';
                
                // Load hacks from Firebase
                loadHacksFromFirebase();
                
                document.getElementById('filesSection').scrollIntoView({
                    behavior: 'smooth'
                });
            } else if (!sessionActive) {
                alert("‚ö†Ô∏è Your session has expired! Please watch 5 ads again.");
                resetProgressSilently();
            } else if (waitingSeconds > 0) {
                // Do nothing - just don't show alert
            } else {
                alert(`‚ö†Ô∏è Watch ${totalAds - adsWatched} more ad${totalAds - adsWatched > 1 ? 's' : ''} to unlock!`);
            }
        }
