        async function trackDownload(hackId) {
            try {
                // Increment download count in Firebase
                const downloadsRef = database.ref(`hacks/${hackId}/downloads`);
                const snapshot = await downloadsRef.once('value');
                const currentDownloads = snapshot.val() || 0;
                
                await downloadsRef.set(currentDownloads + 1);
                
                // Also log download event
                const downloadLogRef = database.ref('download_logs').push();
                await downloadLogRef.set({
                    hack_id: hackId,
                    hack_name: currentDownloadFile?.title || 'Unknown',
                    timestamp: new Date().toISOString(),
                    user_agent: navigator.userAgent
                });
            } catch (error) {
                console.error('Error tracking download:', error);
            }
        }
        
        function checkExpiredHacks() {
            const currentDate = new Date().toISOString().split('T')[0];
            
            hacksList = hacksList.filter(hack => {
                if (!hack.expiry_date) return true;
                return hack.expiry_date >= currentDate;

            }

        function isExpired(expiryDate) {
            if (!expiryDate) return false;
            
            const today = new Date().toISOString().split('T')[0];
            return expiryDate < today;
        }
        
        function getDaysRemaining(expiryDate) {
            if (!expiryDate) return 0;
            
            const today = new Date();
            const expiry = new Date(expiryDate);
            const diffTime = expiry - today;
            const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24));
            
            return diffDays > 0 ? diffDays : 0;
        }

        function initSession() {
            const lastSession = localStorage.getItem('lastSessionData');
            if (lastSession) {
                const sessionData = JSON.parse(lastSession);
                const timeSinceLastActivity = Date.now() - sessionData.lastActivity;

                    resetProgressSilently();
                } else {

                    sessionActive = true;
                }
            }

            saveSession();
        }
